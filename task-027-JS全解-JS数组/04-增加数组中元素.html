            改变原数组

        乱序排列
            arr.sort()
            arr.sort(function(a, b) {
                if (a > b) {
                    return 1
                } else if (a === b) {
                    return 0
                } else {
                    return -1
                }
            })

            arr.sort((a, b) => return a.score - b.score)
