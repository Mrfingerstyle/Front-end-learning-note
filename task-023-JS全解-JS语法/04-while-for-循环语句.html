        for语句 语法糖
            for 是 while的简便写法
            语法 
                for(表达式1,表达式2,表达式3) {
                    // ...
                }
            for (var i = 0; i < 5; i++) {
                setTimeout(() => {
                    console.log(i);
                }, 0)
            }
            // 55555
            // 此处把var 改为 let 就可以得到结果 01234

            var a = 1;
            function fn() {
                console.log(a);
            }
            a = 5;
            setTimeout(fn);
            // 5

            for循环执行完毕 才执行setTimeout
