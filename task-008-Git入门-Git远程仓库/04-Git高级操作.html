        使用 bash alias 简化指令
        touch ~/.bashrc
        echo 'alias ga="git add"'>> ~/.bashrc
        echo 'alias gc="git commit -v"'>> ~/.bashrc
        echo 'alias gl="git pull"'>> ~/.bashrc
        echo 'alias gp="git push"'>> ~/.bashrc
        echo 'alias gco="git checkout"'>> ~/.bashrc
        echo 'alias gst="git status -sb"'>> ~/.bashrc
